4. EXEMPLO EM JAVA

import java.net.URI; import java.net.http.HttpClient; import
java.net.http.HttpRequest; import java.net.http.HttpResponse; import
java.util.Base64; import java.time.Duration;

public class JiraIntegration {

    public static void main(String[] args) throws Exception {

        String email = "<SEU_EMAIL_JIRA>";
        String token = "<SEU_TOKEN_JIRA>";
        String url = "https://<SEU_DOMINIO>.atlassian.net/rest/api/3/issue";

        String auth = email + ":" + token;
        String encodedAuth = Base64.getEncoder().encodeToString(auth.getBytes());
        String authHeader = "Basic " + encodedAuth;

        String jsonBody = """
        {
          "fields": {
            "project": {
              "key": "<CHAVE_DO_PROJETO>"
            },
            "summary": "Alerta gerado automaticamente",
            "description": {
              "type": "doc",
              "version": 1,
              "content": [
                {
                  "type": "paragraph",
                  "content": [
                    {
                      "type": "text",
                      "text": "Mensagem dinâmica aqui"
                    }
                  ]
                }
              ]
            },
            "issuetype": {
              "name": "<TIPO_DO_ISSUE>"
            },
            "assignee": null
          }
        }
        """;

        HttpClient client = HttpClient.newHttpClient();

        HttpRequest request = HttpRequest.newBuilder()
                .uri(new URI(url))
                .header("Authorization", authHeader)
                .header("Content-Type", "application/json")
                .POST(HttpRequest.BodyPublishers.ofString(jsonBody))
                .timeout(Duration.ofSeconds(10))
                .build();

        HttpResponse<String> response = client.send(request, HttpResponse.BodyHandlers.ofString());

        System.out.println("Status Code: " + response.statusCode());
        System.out.println("Response Body: " + response.body());
    }

}
